<script>

const API_URL = "https://script.google.com/macros/s/AKfycbwBr-OhmNqKbqYQtC6TxuLPTnt0ASf8vl0juAm2XgW5t-0tw4q0rXStUKEqhgrowCMT/exec";

 

let signUps = { appetizers: [], sides: [], drinks: [], desserts: [] };

 

document.addEventListener('DOMContentLoaded', function() {

loadSignUps();

});

 

async function addItem() {

const name = document.getElementById('nameInput').value.trim();

const category = document.getElementById('categorySelect').value;

const item = document.getElementById('itemInput').value.trim();

 

if (!name || !category || !item) {

alert('Please fill in all fields!');

return;

}

 

// Send to Google Sheets backend

await fetch(API_URL, {

method: "POST",

body: JSON.stringify({ name, category, item })

});

 

// Clear form

document.getElementById('nameInput').value = '';

document.getElementById('categorySelect').value = '';

document.getElementById('itemInput').value = '';

 

// Reload list

loadSignUps();

showSuccessMessage(name, item, category);

}

 

async function loadSignUps() {

const res = await fetch(API_URL);

const data = await res.json();

 

// Reset signUps

signUps = { appetizers: [], sides: [], drinks: [], desserts: [] };

 

data.forEach(entry => {

if (signUps[entry.category]) {

signUps[entry.category].push(entry);

}

});

 

updateDisplay();

}

 

function updateDisplay() {

updateList('appetizers', 'appetizersList', 'appetizersCount', 'totalAppetizers');

updateList('sides', 'sidesList', 'sidesCount', 'totalSides');

updateList('drinks', 'drinksList', 'drinksCount', 'totalDrinks');

updateList('desserts', 'dessertsList', 'dessertsCount', 'totalDesserts');

updateTotalCount();

}

 

function updateList(category, listId, countId, totalId) {

const list = document.getElementById(listId);

const items = signUps[category];

if (items.length === 0) {

list.innerHTML = `<div class="text-gray-500 italic">No ${category} signed up yet</div>`;

} else {

list.innerHTML = items.map(item => `

<div class="flex justify-between items-center bg-gray-50 rounded-lg p-3 border-l-4 border-gray-300">

<div>

<div class="font-semibold text-gray-800">${item.item}</div>

<div class="text-sm text-gray-600">by ${item.name}</div>

</div>

</div>

`).join('');

}

document.getElementById(countId).textContent = items.length;

document.getElementById(totalId).textContent = items.length;

}

 

function updateTotalCount() {

const total = signUps.appetizers.length + signUps.sides.length + signUps.drinks.length + signUps.desserts.length;

document.getElementById('totalItems').textContent = total;

}

 

function showSuccessMessage(name, item, category) {

const message = document.createElement('div');

message.className = 'fixed top-4 right-4 bg-green-500 text-white px-6 py-3 rounded-lg shadow-lg z-50';

message.innerHTML = ` ${name} added "${item}" to ${category}!`;

document.body.appendChild(message);

setTimeout(() => {

message.remove();

}, 3000);

}

 

// Allow Enter key to submit

document.addEventListener('keypress', function(e) {

if (e.key === 'Enter') {

addItem();

}

});
